    $.get('/stat/factoryCharts.do').done(function (data) {

                trigger: 'item',

            },
            series : [
                {
                    name: '访问来源',
                    type: 'pie',
                    radius : '65%',
                    center: ['50%', '50%'],
                    data:data,
                    itemStyle: {
                        emphasis: {
                            shadowBlur: 10,
                            shadowOffsetX: 0,
                            shadowColor: 'rgba(0, 0, 0, 0.5)'
                        }
                    }
                }
            ]
        })
